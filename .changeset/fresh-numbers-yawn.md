  <div class='iui-input-with-icon'>
    <div
      role='combobox'
      tabindex='0'
      class='iui-select-button iui-field'
+     data-iui-multi
    >
-     <span class='iui-content'>
-       <div class='iui-select-tag-container'>…</div>
-     </span>
    </div>
    <svg class='iui-end-icon'>…</svg>
+   <span class='iui-content'>
+     <div class='iui-select-tag-container'>…</div>
+   </span>
